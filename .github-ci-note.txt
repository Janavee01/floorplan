# GitHub Actions CI — save this as .github/workflows/test.yml on your repo

name: tests
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"
      - name: Install system deps
        run: sudo apt-get install -y tesseract-ocr
      - name: Install Python deps
        run: pip install opencv-python numpy pytesseract Pillow pyyaml pytest
      - name: Run tests
        run: pytest tests/ -v
